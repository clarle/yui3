Code Sample

<script src="http://yui.yahooapis.com/3.3.0/build/yui/yui-min.js"></script>
YUI({
    //Last Gallery Build of this module
    gallery: 'gallery-2011.02.02-21-07'
}).use('gallery-xarno-calendar', 'gallery-xarno-calendar-controls', function(Y) {
 
  var c = new Y.Xarno.Calendar();
  c.plug(Y.Plugin.Xarno.CalendarControls, {position:'header'});
  c.render();
 
});
