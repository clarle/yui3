Code Sample

YUI().use('gallery-querystring', function(Y) {
     var qs  = new Y.querystring();  
     var key = 'foo';  
     var mes = '';  
     if(qs.contains(key)){  
         mes = 'value is ' + qs.get(key);
     }  
     else{  
         mes = 'not exist';
     }  
     alert(mes);
});
