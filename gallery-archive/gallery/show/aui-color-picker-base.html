A color picker widget that provides the ability to choose a color with sliders and present the selection in either RGB, HEX, or HSV values.
The ColorPicker is shown in an overlay that is triggered, and it is dismissed when a user clicks outside of it.

You can create a color picker like so:

Code Sample

<script src="http://yui.yahooapis.com/3.2.0/build/yui/yui-min.js"></script>
YUI({
    //Last Gallery Build of this module
    gallery: 'gallery-2011.02.09-21-32',
    modules: {
        'gallery-aui-skin-base': {
            fullpath: 'http://yui.yahooapis.com/gallery-2011.02.09-21-32/build/gallery-aui-skin-base/css/gallery-aui-skin-base-min.css',
            type: 'css'
        },
        'gallery-aui-skin-classic': {
            fullpath: 'http://yui.yahooapis.com/gallery-2011.02.09-21-32/build/gallery-aui-skin-classic/css/gallery-aui-skin-classic-min.css',
            type: 'css',
            requires: ['gallery-aui-skin-base']
        }
    }
}).use('gallery-aui-color-picker-base', function(Y) {
// Basic ColorPicker
		var colorPicker1 = new Y.ColorPicker().render();
 
		// OR To update an input when the color changes:
		var colorPicker2 = new Y.ColorPicker({
			after: {
				colorChange: function(event) {
					Y.one('#input1').val(this.get('hex')); // Update an input field with the hex value
				}
			}
		}).render();
 
});
