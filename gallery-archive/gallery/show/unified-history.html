Provides HistoryHash capability to support path-like fragment pattern.

YUI3 HistoryHash module support only key-value pair hash fragment by default. This module makes it display path-like hash fragment by adding usePath config.

Code Sample

<script src="http://yui.yahooapis.com/3.3.0/build/yui/yui-min.js"></script>
YUI({
    //Last Gallery Build of this module
    gallery: 'gallery-2011.04.13-22-38'
}).use('gallery-unified-history', function(Y) {
 
    var history = new Y.HistoryHash({
       usePath: ['page', 'display']  
    });
 
    // By default, below statement change current page URL to be "...#page=12345&display=grid".
    // With gallery-unified-history loaded, the URL will be "#12345/grid".
    history.add({page: '12345', display:'grid'});
 
    history.on('pageChange', function(ev) {
          // This is same as default HistoryHash. It can just parse hash fragment like "page=12345&display=grid"
    });
 
});
