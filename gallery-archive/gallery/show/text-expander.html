This code attaches to textarea nodes via the 'plug' method. You can specify the min_height, max_height, and line_height, if wanted.

Code Sample

<script src="http://yui.yahooapis.com/3.0.0/build/yui/yui-min.js"></script>
YUI({
    //Last Gallery Build of this module
    gallery: 'gallery-2010.03.23-17-54'
}).use("node", 'gallery-text-expander', function(Y) {
    Y.all("textarea").plug(Y.TextExpander);
});
