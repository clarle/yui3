Return to Row Expansion for DataTable

YUI({
    //Last Gallery Build of this module
    gallery: 'gallery-2014.03.06-14-38'
}).use('gallery-datatable-row-expansion', function(Y)
{
	var table = new Y.DataTable(
	{
		columns:
		[
			{ key: Y.Plugin.RowExpansion.column_key, label: 'Â ' },
			{ key: 'title', label: 'Title' },
			{ key: 'year', label: 'Year' }
		]
	});
	table.plug(Y.Plugin.RowExpansion,
	{
		uniqueIdKey: 'id',
		template: function(data)
		{
			return Y.Lang.sub('Author: {author}, Length of title: {len}',
			{
				author: data.author,
				len:    data.title.length
			});
		}
	});
	table.render('#my-table');
});
